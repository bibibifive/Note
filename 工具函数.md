# 工具函数

### 防抖

```js
function thro(fn, delay = 100) {
  let timer = null
  return function () {
    if (timer) clearTimeout(timer)
    timer = setTimeout(() => {
      fn.apply(this, arguments)
      let timer = null
    }, delay)
  }
}
```

### 节流

```js
function throttle(fn, gap = 100) {
  let timer = null
  return function () {
    if (timer) return
    timer = setTimeout(() => {
      fn.apply(this, arguments)
      let timer = null
    }, gap)
  }
}
```
